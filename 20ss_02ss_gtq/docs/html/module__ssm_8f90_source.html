234

236  do ix=1,nbox(1)
237  do iy=1,nbox(2)
238 
239  write(*,*) ix,iy
240 
241  centre=cmplx(x0+(ix-1)*yoko, y0+(iy-1)*tate,kind(1.d0))
242 
243  sft(1)=centre-ione*0.5d0*tate
244  sft(2)=centre+0.5d0*yoko
245  sft(3)=centre+ione*0.5d0*tate
246  sft(4)=centre-0.5d0*yoko
247 
248  st=centre-0.5d0*yoko-0.5d0*ione*tate
249  write(9,*) real(st),aimag(st)
250  st=st+yoko
251  write(9,*) real(st),aimag(st)
252  st=st+tate*ione
253  write(9,*) real(st),aimag(st)
254  st=st-yoko
255  write(9,*) real(st),aimag(st)
256  st=st-ione*tate
257  write(9,*) real(st),aimag(st)
258  write(9,*); write(9,*)

261  mhat(:,:,:)=zero
262  h(:,:)=zero
263  sh(:,:)=zero

265  !
266  ! A(z) ymat = vmatを解いて、ymatを作成し、モーメントshat=\int z^k ymat dz を計算
267  !
268  do i=1, 4 !1:→, 2:↑, 3:←, 4:↓
269  do iskbn=1,ngt(i)
270 
271  z=ad(scl(i)*gt(ngt(i))%s(ns)%gz(iskbn)+sft(i))
272 
273  call assemble_matrix(z) !Aを計算
274  fmat=amat%f(0)
275  call lapack_lu(m,fmat,ipiv) !AをLU分解
276 
277  ymat(:,:)%f(0)=vmat(:,:)
278  do j=0,2*ns
279  if(j.ne.0) ymat(:,:)%f(j)=zero
280  do k=0,j-1
281  ymat(:,:)%f(j)=ymat(:,:)%f(j)-choose(j)%a(k)*matmul(amat(:,:)%f(j-k),ymat(:,:)%f(k))
282  end do
283  call lapack_solve(m,nblck,fmat,ipiv,ymat%f(j))
284  end do
285 
286  do k=0,2*nhank-1
287  cf=scl(i)
288  do is=0,2*ns
289  shat(:,:,k)=shat(:,:,k)+ymat(:,:)%f(is)*gt(ngt(i))%s(ns)%we(is,iskbn)*cf
290  cf=cf*scl(i)
291  end do
292  ymat(:,:)=ymat(:,:)*(z-centre)/taikaku
293  end do
294 
295  end do
296  end do
297  shat(:,:,:)=shat(:,:,:)/(2.d0*pi*ione)
298 
299  !

303  mhat(:,:,k)=matmul(transpose(umat),shat(:,:,k))
304  end do

306 
307  !

309  !

311  do j=1,nhank
312  itmp=0

314  h(itmp+1:itmp+nblck,jtmp+1:jtmp+nblck)=mhat(:,:,i+j-2)
315  sh(itmp+1:itmp+nblck,jtmp+1:jtmp+nblck)=mhat(:,:,i+j-1)
316  itmp=itmp+nblck
317  end do
318  jtmp=jtmp+nblck
319  end do
320 
321 
322  nev_tmp=0
323 
324  !
325  ! Hankel行列のフロベニウスノルム/要素数が小さければ固有値なし
326  !

329  end if

338  stop
339  end if

341  ! ! 特異値と最大特異値で正規化された特異値を書き出す
342  ! open(97,file="sgm.dat",status="replace")
343  ! do i=1,n
344  ! write(97,*) sgm(i), sgm(i)/sgm(1)
345  ! end do
346  ! close(97)
347 
348  ! i+1番目の特異値/i番目の特異値がsgm_thres以下なら，固有値の数=i

351  nev_tmp=i
352  end if
353  end do
354 
355  ! 最大特異値が小さければ固有値がないとする (↑の判定だけだとゴミがたくさん引っかかる場合)
356  if(sgm(1).lt.zero_thres) then
357  nev_tmp=0
358  end if
359 
360  if(nev_tmp.eq.0) cycle
361 
362  write(*,*) nev_tmp,"eigen values are found"
363  !
364  ! 一般化固有値問題を解くと、もとの問題の固有値が求まる
365  !
366  allocate(alp(nev_tmp),bet(nev_tmp),vr(nev_tmp,nev_tmp))
367  call solve_generalized_eigen_x_2(nev_tmp,sh(1:nev_tmp,1:nev_tmp),h(1:nev_tmp,1:nev_tmp),alp,bet,vr,info)
368  if(info.ne.0) cycle
369 
370  !
371  ! 固有ベクトル
372  !
373  allocate(stmp(m,nev_tmp))
374  itmp=0
375  do j=0,2*nhank-1
376  do i=1,nblck
377  itmp = itmp + 1
378  stmp(:,itmp)=shat(:,i,j)
379  if(itmp.eq.nev_tmp) exit
380  end do
381  if(itmp.eq.nev_tmp) exit
382  end do
383  allocate(evv_tmp(m,nev_tmp))
384  evv_tmp=matmul(stmp,vr)
385  deallocate(stmp)

388  ! 固有値を格納 (積分路の内側にあるもののみ)
389  !
390  nev_new = count(abs(alp(:)/bet(:))<1.d0) !abs(alp(:)/bet(:))<1.d0を満たす固有値の数
391 
392  allocate(ev_new(nev_new+nev))
393  allocate(evv_new(m,nev_new+nev))
394 
395  if(nev.ne.0)then ! 既に持っていた固有対の配列
396  ev_new(1:nev)=ev(:)
397  evv_new(:,1:nev)=evv(:,:)
398  end if
399  itmp=nev
400  do i=1,nev_tmp
401  if(abs(alp(i)/bet(i))<1.d0) then
402  itmp = itmp + 1
403  ev_new(itmp) = alp(i)/bet(i)*taikaku+centre
404  evv_new(:,itmp) = evv_tmp(:,i)
405  end if
406  end do
407 
408  ! 固有対を更新
409  if(allocated(ssm_%ev)) deallocate(ssm_%ev)
410  if(allocated(ssm_%evv)) deallocate(ssm_%evv)
411  if(associated(ev)) nullify(ev)
412  if(associated(evv)) nullify(evv)
413  nev=nev+nev_new
414  allocate(ssm_%ev(nev)); ev=>ssm_%ev
415  allocate(ssm_%evv(m,nev)); evv=>ssm_%evv
416  ev(:)=ev_new(:)
417  evv(:,:)=evv_new(:,:)
418 
419  deallocate(alp,bet,vr,evv_tmp,ev_new,evv_new)
420 
421  ! 固有ベクトルは絶対値最大の要素の絶対値が1になるように正規化しておく
422  do i=1,nev
423  ztmp=one/maxval(sqrt(real(evv(:,i))**2+aimag(evv(:,i))**2))
424  evv(:,i)=evv(:,i)*ztmp
425  end do
426 
427  end do
428  end do
429  close(9)
