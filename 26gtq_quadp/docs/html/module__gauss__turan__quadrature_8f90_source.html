148  subroutine assemble_gauss_turan
149  integer :: i, j, k, l, is, n, ngl, ig, ipiv(maxngt-1), info
150  real(16) :: bet(maxngt-1), w(maxngt-1), jac(maxngt-1,maxngt-1), f(maxngt-1)
151  real(16) :: p(0:maxngt), q(0:maxngt,0:maxngt), b(0:maxngt,0:maxngt)
152  real(16) :: td(maxngt), tl(maxngt-1), work(2*maxngt-2)
153  complex(16) :: ev(maxngt,maxngt)
154 
155  real(16) :: ahat(2*maxs+1,2*maxs+1), u(2*maxs), muhat(0:2*maxs), tmp, bb(2*maxs+1)
156 
157  ! GLを準備
158  !$omp parallel do private(i) schedule(dynamic)
159  do i=1,maxngl
160  write(0,*) "GL",i,"/",maxngl
161  gl(i)%ng=i
162  allocate(gl(i)%gz(i),gl(i)%we(i))
163  call assemble_gauss_legendre(i,gl(i)%gz,gl(i)%we)
164  end do
165  !$omp end parallel do
166 
167  ! 分点数をset
168  do is=0,maxs
169  do i=1,maxngt
170  gt(i)%s(is)%ng=i
171  end do
172  end do
173 
174  ! s=0解微分を使うGauss-TuranはGauss-Legendreに他ならない
175  do i=1,maxngt
176  gt(i)%s(0)%gz(:)=gl(i)%gz(:)
177  gt(i)%s(0)%we(0,:)=gl(i)%we(:)
178  end do
179 
180  do is=1,maxs !is=(1,..,maxs)階微分を使うGauss-Turan多項式の零点を探す
181  write(0,*) "GT (node)",is,"/",maxs
182 
183  ! Turan多項式の満たす漸化式の係数(beta)をnewton法で求める
184  ! P(n;-1)(x)=0
185  ! P(n;0)(x)=1
186  ! P(n;1)(x)=x
187  ! P(n;k+1)(x)=xP(n;k)(x)-beta(k)P(n;k-1)(x)
188 
189  ! n=2のときのNewton法の初期値
190  bet(1)=(2.q0*is+1.q0)/(2.q0*is+3.q0)
191  bet(3:)=0.q0
192 
193  do n=2,maxngt ! n=2次公式から順番に
194  write(0,*) "GT (node)",is,"/",maxs, n,"/",maxngt
195  ngl=(is+1)*n ! ngl点のGL公式でfの値を計算し
196  if(n.gt.2) bet(n-1)=bet(n-2)
197  do ! Newton法で
198  f(:)=0.q0 ! f(beta)=0となるbetaを探す
199  jac(:,:)=0.q0 ! <=Newton法で使うJacobi行列
200  ! fとそのJacobianを計算
201  !$omp parallel do private(ig,i,p,j,b,q) reduction(+:f,jac)
202  do ig=1,ngl
203  call comp_poly_and_dpoly(n,bet(1:n-1),gl(ngl)%gz(ig),p(0:n),b(0:n,0:n))
204  ! do i=0,n
205  ! p(i)=poly(n,i,bet,gl(ngl)%gz(ig))
206  ! do j=0,n
207  ! b(i,j)=dpoly(n,i,j,bet,gl(ngl)%gz(ig))
208  ! end do
209  ! end do
210  do i=0,n
211  do j=0,n
212  q(i,j)=p(i)*(b(i,j)*p(n)+is*b(n,j)*p(i))
213  end do
214  end do
215  do i=1,n-1
216  f(i)=f(i)+(bet(i)*p(i-1)**2-p(i)**2)*p(n)**(2*is)*gl(ngl)%we(ig)
217  do j=1,n-1
218  jac(i,j)=jac(i,j)+2.q0*(bet(i)*q(i-1,j)-q(i,j)&
219  +0.5q0*delta(i,j)*p(i-1)**2*p(n))*p(n)**(2*is-1)*gl(ngl)%we(ig)
220  end do
221  end do
222  end do
223  !$omp end parallel do
224  ! Newton法の更新量を求める
225  w=f
226  call lu_and_solve(n-1,jac(1:n-1,1:n-1),w(1:n-1),ipiv(1:n-1))
227  write(0,*) sqrt(dot_product(f(1:n-1),f(1:n-1)))
228  if(sqrt(dot_product(f(1:n-1),f(1:n-1))).le.epsilon(1.q0)) exit
229  bet(:)=bet(:)-w(:) ! 更新
230  end do
231 
232  td(:)=0.q0
233  do i=1,n-1
234  tl(i)=sqrt(bet(i))
235  end do
236 
237  ev(:,:)=cmplx(0.q0,0.q0,kind(1.q0))
238  do i=1,n
239  ev(i,i)=cmplx(1.q0,0.q0,kind(1.q0))
240  end do
241 
242  call qzsteqr("V",n,td(1:n),tl(1:n-1),ev(1:n,1:n),n,work(1:n),info)
243  ! write(*,*) "info", info
244 
245  do i=1,n
246  gt(n)%s(is)%gz(i)=td(i)
247  end do
248 
249  end do
250 
251  ! n=1の時は分点は零
252  gt(1)%s(is)%gz(1)=0.q0
253 
254  end do
255 
256  do is=1,maxs ! 2*is階微分を使うGT公式の
257  write(0,*) "GT (weight)",is,"/",maxs
258  do n=1,maxngt ! n次の公式の
259  do j=1,n ! j番分点の重みを計算する
260 
261  ! uを計算
262  u(:)=0.q0
263  do i=1,2*is
264  do l=1,n
265  if(l.ne.j) then
266  u(i)=u(i)+(gt(n)%s(is)%gz(l)-gt(n)%s(is)%gz(j))**(-i)
267  end if
268  end do
269  end do
270 
271  ! ahatを計算
272  ahat(:,:)=0.q0
273  do k=1,2*is+1
274  ahat(k,k)=1.q0
275  end do
276  do l=1,2*is
277  do k=1,2*is+1-l
278  do i=1,l
279  ahat(k,k+l)=ahat(k,k+l)-(2.q0*is+1.q0)/dble(l)*u(i)*ahat(i,l)
280  end do
281  end do
282  end do
283 
284  ! muhatを計算
285  ngl=(is+1)*n
286  muhat(:)=0.q0
287  do k=0,2*is
288  do ig=1,ngl
289  tmp=1.q0
290  do l=1,n
291  if(l.ne.j) then
292  tmp=tmp*((gl(ngl)%gz(ig)-gt(n)%s(is)%gz(l))&
293  /(gt(n)%s(is)%gz(j)-gt(n)%s(is)%gz(l)))**(2*is+1)
294  end if
295  end do
296  muhat(k)=muhat(k)+(gl(ngl)%gz(ig)-gt(n)%s(is)%gz(j))**k*tmp*gl(ngl)%we(ig)
297  end do
298  end do
299 
300  ! bを計算
301  bb(:)=0.q0
302  bb(2*is+1)=muhat(2*is)
303  do k=2*is,1,-1
304  bb(k)=muhat(k-1)
305  do l=k+1,2*is+1
306  bb(k)=bb(k)-ahat(k,l)*bb(l)
307  end do
308  end do
309 
310  ! bb-->weight
311  do k=1,2*is+1
312  gt(n)%s(is)%we(k-1,j)=bb(k)/factorial(k-1)
313  end do
314 
315  end do
316  end do
317  end do
318 
319  ! 分点・重みをcheck
320  do is=0,maxs
321  write(*,*) is, "s"
322  do n=1,maxngt
323  write(*,*) n, "n"
324  do i=1,n
325  write(*,*) dble(gt(n)%s(is)%gz(i)), dble(gt(n)%s(is)%we(0:2*is,i)), i
326  end do
327  end do
328  end do
329 
330  ! GLを捨てる
331  do i=1,maxngl
332  if(allocated(gl(i)%gz)) deallocate(gl(i)%gz)
333  if(allocated(gl(i)%we)) deallocate(gl(i)%we)
334  end do
335 
336  end subroutine assemble_gauss_turan
